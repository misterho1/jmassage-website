- [ ] **Step 1: Scaffold the script**

```python
"""Guarded J Massage price restructure (2026-06-13). Aborts on any count mismatch."""
import sys, pathlib
ROOT = pathlib.Path(__file__).resolve().parent.parent

# file -> list of (old, new, expected_count)
REPLACEMENTS = {
    # populated per file in Step 2, e.g. for swedish-massage.html:
    # "services/swedish-massage.html": [
    #     ("From $55", "From $85", 4),
    #     ('"price": "55"', '"price": "85"', 1),
    #     ... etc
    # ],
}

def apply(check_only=False):
    total = 0
    for rel, rules in REPLACEMENTS.items():
        p = ROOT / rel
        text = p.read_text(encoding="utf-8")
        for old, new, n in rules:
            found = text.count(old)
            assert found == n, f"{rel}: expected {n}x {old!r}, found {found}"
            text = text.replace(old, new)
            total += n
        if not check_only:
            p.write_text(text, encoding="utf-8", newline="\n")
    print(f"{'CHECK' if check_only else 'APPLIED'}: {total} replacements across {len(REPLACEMENTS)} files")

if __name__ == "__main__":
    apply(check_only="--check" in sys.argv)
```
